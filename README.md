## Usage

```js
// ES6
import {getNum, getStr} from '@islishude/get-random'
import Random from '@islishude/get-random'
// Node
var Random = require('@islishude/get-random')

// get random number of given range
Random.getNum(2,100)
// get random string,default length is 6
Random.getStr(6)
```
